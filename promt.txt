            <script>
                function replicateProduct(productId) {
                    // Make an API call to replicate the product with the given ID
                    fetch(`/branch/catalogue/replicate/${productId}`, {
                        method: 'POST',
                        headers: {
                            'Content-Type': 'application/json',
                            },
                    })
                    .then(response => response.json())
                    .then(data => {
                        console.log('API Response:', data);
                
                        // Check if the _id field is present in the data object
                        if (data._id) {
                            const replicatedProductId = data._id;
                            console.log('Replicated Product ID:', replicatedProductId);
                        } else {
                            console.log('Replicated Product ID not found in API response.');
                        }
                
                        // You can further inspect the response to determine its format and structure
                    })
                    .catch(error => {
                        console.error('Error replicating data:', error);
                    });
                }
                        
                // Attach event listeners to each toggle switch
                const toggleSwitches = document.querySelectorAll('input[type="checkbox"]');
                toggleSwitches.forEach(toggleSwitch => {
                    toggleSwitch.addEventListener('change', () => {
                        if (toggleSwitch.checked) {
                            const productId = toggleSwitch.getAttribute('data-product-id');
                            replicateProduct(productId);
                        }
                    });
                });
            </script>

        $(document).ready(function () {
          // Handle click event on the "Delete" link
          $('a[data-bs-toggle="modal"]').on('click', function () {
            // Get the Product ID from the link's href attribute
            const productId = this.href.split('/').pop();
      
            // Update the "Yes" button's data attribute with the category ID for reference when confirming deletion
            $('#confirmDeleteBtn').data('category-id', productId);
          });
      
          // Handle click event on the "Yes" button inside the modal
          $('#confirmDeleteBtn').on('click', function () {
            // Get the category ID from the data attribute
            const productId = $(this).data('category-id');
      
            // Make an AJAX request to the server to delete the category
            $.ajax({
              method: 'DELETE',
              url: `/admin/product/delete/${productId}`,
              success: function (data) {
                // If the deletion is successful, you can handle the response or reload the page
                console.log(data); // Optional: Log the server response
                window.location.reload(); // Reload the page to reflect the changes
                },
              error: function (error) {
                // Handle errors if any
                console.error(error);
                },
            });
          });
        });
      </script>
